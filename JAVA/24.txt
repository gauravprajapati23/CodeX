1. Write a java program to scroll the text from left to right continuously.         
[15 M] 
// Q1
import javax.swing.*;

public class ScrollText extends JFrame implements Runnable {
    JLabel label;
    String text = "Scrolling Text Demo     ";
    Thread t;

    public ScrollText() {
        label = new JLabel(text, SwingConstants.CENTER);
        add(label);

        setSize(400,200);
        setVisible(true);
        setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE);

        t = new Thread(this);
        t.start();
    }

    public void run() {
        try {
            while(true){
                text = text.substring(1) + text.charAt(0);
                label.setText(text);
                Thread.sleep(200);
            }
        } catch(Exception e){}
    }

    public static void main(String[] args) {
        new ScrollText();
    }
}
