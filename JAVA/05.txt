1. Write a Java Program to create the hash table that will maintain the mobile number and 
student name. Display the details of student using Enumeration interface. 
[15 M] 
// Q1
import java.util.*;

public class StudentMobile {
    public static void main(String[] args) {
        Hashtable<String, String> ht = new Hashtable<>();

        ht.put("9876543210", "Rahul");
        ht.put("9123456780", "Priya");
        ht.put("9988776655", "Amit");

        Enumeration<String> keys = ht.keys();

        while (keys.hasMoreElements()) {
            String mobile = keys.nextElement();
            String name = ht.get(mobile);
            System.out.println("Mobile: " + mobile + " Name: " + name);
        }
    }
}
